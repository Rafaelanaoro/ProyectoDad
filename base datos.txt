CREATE TABLE Usuarios  
(  
    id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,  
    nombre VARCHAR(50),
    apellido VARCHAR(50),
    valor DOUBLE,
    groupid INT,
	 fecha DATE  
)


CREATE TABLE LED  
(  
    idled INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
    nivel_luz DOUBLE,
    groupid INT,
	 fecha DATE,
	 estado BOOLEAN  
)


CREATE TABLE Sensor  
(  
    idNFC INT NOT NULL AUTO_INCREMENT PRIMARY KEY,  
    valor DOUBLE,
    groupid INT,
	 fecha DATE,
	 estado BOOLEAN  
)


CREATE TABLE Servo
(  
    idServo INT NOT NULL AUTO_INCREMENT PRIMARY KEY,  
    estado BOOLEAN,
    valor INT,
    groupId INT,
	 fecha DATE  
)
